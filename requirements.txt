# ── Local Development ─────────────────────────────────────────────────────────
# pip install -r requirements.txt
# Install poppler (required by pdf2image):
#   macOS:          brew install poppler
#   Ubuntu/Debian:  sudo apt-get install -y poppler-utils
# Optional Redis (for async evaluation):
#   macOS:          brew install redis && brew services start redis
#   Ubuntu/Debian:  sudo apt-get install -y redis-server

# ── Docker ────────────────────────────────────────────────────────────────────
# Add to Dockerfile:
#   RUN apt-get update && apt-get install -y poppler-utils && rm -rf /var/lib/apt/lists/*

# ── Vercel (Python Serverless) ────────────────────────────────────────────────
# 1. Set env vars in Vercel dashboard:
#      OPENAI_API_KEY, MONGO_URI, SMTP_HOST, SMTP_USER, SMTP_PASS, SMTP_PORT
#      REDIS_URL (optional — for async bulk evaluation)
# 2. /tmp is the only writable directory (IS_VERCEL=1 auto-detects this).
# 3. poppler-utils: Vercel Lambda does NOT support apt-get.
#    For Vercel deployment, pdf2image/poppler won't work.
#    The app gracefully falls back to GPT-4o vision-only OCR (Path C).
#    For full image-diff OCR support, use Railway, Render, or Docker instead.
# 4. Build React frontend first:  cd vidyai-react && npm run build
# 5. Point Flask static_folder to vidyai-react/dist/

# ── Railway / Render / Fly.io ─────────────────────────────────────────────────
# These platforms support custom buildpacks. Add a Nixpacks config or Dockerfile:
#   apt-get install -y poppler-utils
# All features including image-diff OCR work fully on these platforms.
